# –≥–ª–∞–≤–Ω–∞—è —Ñ—É–Ω–∫—Ü–∏—è –ø—Ä–æ–≥—Ä–∞–º–º—ã 
def sum_odd(matrix):
    """
    –í–æ–∑–≤—Ä–∞—â–∞–µ—Ç —Å–ø–∏—Å–æ–∫ —Å—É–º–º –Ω–µ—á–µ—Ç–Ω—ã—Ö —ç–ª–µ–º–µ–Ω—Ç–æ–≤ –¥–ª—è –∫–∞–∂–¥–æ–π —Å—Ç—Ä–æ–∫–∏ –º–∞—Ç—Ä–∏—Ü—ã
    """
    result = []
    for row in matrix:
        row_sum = 0
        for element in row:
            if element % 2 != 0:  # –ü—Ä–æ–≤–µ—Ä—è–µ–º, –Ω–µ—á–µ—Ç–Ω–æ–µ –ª–∏ —á–∏—Å–ª–æ
                row_sum += element
        result.append(row_sum)
    return result

M = int(input("–í–≤–µ–¥–∏—Ç–µ –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ —Å—Ç—Ä–æ–∫ M: "))
N = int(input("–í–≤–µ–¥–∏—Ç–µ –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ —Å—Ç–æ–ª–±—Ü–æ–≤ N: "))

matrix = []
print(f"\n–í–≤–µ–¥–∏—Ç–µ –º–∞—Ç—Ä–∏—Ü—É {M}x{N} (–ø–æ {N} —á–∏—Å–µ–ª –≤ –∫–∞–∂–¥–æ–π —Å—Ç—Ä–æ–∫–µ):")

#–≤–≤–æ–¥ –º–∞—Ç—Ä–∏—Ü—ã 
for i in range(M):
    while True:
        try:
            row_input = input(f"–°—Ç—Ä–æ–∫–∞ {i+1}: ")
            row = list(map(int, row_input.split()))
            
            if len(row) != N:
                print(f"–û—à–∏–±–∫–∞! –ù—É–∂–Ω–æ –≤–≤–µ—Å—Ç–∏ —Ä–æ–≤–Ω–æ {N} —á–∏—Å–µ–ª. –ü–æ–ø—Ä–æ–±—É–π—Ç–µ —Å–Ω–æ–≤–∞.")
                continue
                
            matrix.append(row)
            break
        except ValueError:
            print("–û—à–∏–±–∫–∞! –í–≤–æ–¥–∏—Ç–µ —Ç–æ–ª—å–∫–æ —Ü–µ–ª—ã–µ —á–∏—Å–ª–∞. –ü–æ–ø—Ä–æ–±—É–π—Ç–µ —Å–Ω–æ–≤–∞.")

sums = sum_odd(matrix)

# –≤—ã–≤–æ–¥ —Ä–µ–∑—É–ª—å—Ç–∞—Ç–æ–≤
print("\n–ú–∞—Ç—Ä–∏—Ü–∞:")

    print(row)

suum = 0

for i, row_sum in enumerate(sums, 1):
    suum += row_sum
    print(f"–°—Ç—Ä–æ–∫–∞ {i}: {row_sum}")
print('–û–±—â–∞—è —Å—É–º–º–∞ –ø–æ –≤—Å–µ–º —Å—Ç—Ä–æ–∫–∞–º:', suum)
