//Данная программа находит максимальный среди минимальных элементов строк матрицы

#include <iostream>
using namespace std;

int main() {
    int M, N, matrix[10][10];
    
    cin >> M >> N;
    
    // Ввод матрицы
    for (int i = 0; i < M; i++) {
        for (int j = 0; j < N; j++) {
            cin >> matrix[i][j];
        }
    }
    
    int maxOfMins = -1000000; // начальное значение
    
    for (int i = 0; i < M; i++) {
        int minInRow = matrix[i][0];
        
        // Находим минимальный в строке
        for (int j = 1; j < N; j++) {
            if (matrix[i][j] < minInRow) {
                minInRow = matrix[i][j];
            }
        }
        
        // Сравниваем с максимальным из минимальных
        if (minInRow > maxOfMins || i == 0) {
            maxOfMins = minInRow;
        }
    }
    
    cout << maxOfMins;
}
