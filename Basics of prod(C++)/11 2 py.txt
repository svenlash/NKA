# –î–∞–Ω–Ω–∞—è –ø—Ä–æ–≥—Ä–∞–º–º–∞ –Ω–∞—Ö–æ–¥–∏—Ç –º–∞–∫—Å–∏–º–∞–ª—å–Ω—ã–π —Å—Ä–µ–¥–∏ –º–∏–Ω–∏–º–∞–ª—å–Ω—ã—Ö —ç–ª–µ–º–µ–Ω—Ç–æ–≤ —Å—Ç—Ä–æ–∫ –º–∞—Ç—Ä–∏—Ü—ã

print('–î–∞–Ω–∞ –º–∞—Ç—Ä–∏—Ü–∞ —Ä–∞–∑–º–µ—Ä–∞ ùëÄ √ó ùëÅ. –ù–∞–π—Ç–∏ –º–∞–∫—Å–∏–º–∞–ª—å–Ω—ã–π —Å—Ä–µ–¥–∏ –º–∏–Ω–∏–º–∞–ª—å–Ω—ã—Ö —ç–ª–µ–º–µ–Ω—Ç–æ–≤ –µ–µ —Å—Ç—Ä–æ–∫')

# –≥–ª–∞–≤–Ω–∞—è —Ñ—É–Ω–∫—Ü–∏—è –ø—Ä–æ–≥—Ä–∞–º–º—ã 
def find_max_of_mins(matrix):
    """
    –í–æ–∑–≤—Ä–∞—â–∞–µ—Ç –º–∞–∫—Å–∏–º–∞–ª—å–Ω—ã–π —ç–ª–µ–º–µ–Ω—Ç —Å—Ä–µ–¥–∏ –º–∏–Ω–∏–º–∞–ª—å–Ω—ã—Ö —ç–ª–µ–º–µ–Ω—Ç–æ–≤ –∫–∞–∂–¥–æ–π —Å—Ç—Ä–æ–∫–∏ –º–∞—Ç—Ä–∏—Ü—ã
    """
    # –ù–∞—Ö–æ–¥–∏–º –º–∏–Ω–∏–º–∞–ª—å–Ω—ã–π —ç–ª–µ–º–µ–Ω—Ç –¥–ª—è –∫–∞–∂–¥–æ–π —Å—Ç—Ä–æ–∫–∏
    min_elements = [min(row) for row in matrix]
    
    # –ù–∞—Ö–æ–¥–∏–º –º–∞–∫—Å–∏–º–∞–ª—å–Ω—ã–π —Å—Ä–µ–¥–∏ –Ω–∞–π–¥–µ–Ω–Ω—ã—Ö –º–∏–Ω–∏–º–∞–ª—å–Ω—ã—Ö —ç–ª–µ–º–µ–Ω—Ç–æ–≤
    max_of_mins = max(min_elements)
    
    return max_of_mins, min_elements

M = int(input("–í–≤–µ–¥–∏—Ç–µ –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ —Å—Ç—Ä–æ–∫ M: "))
N = int(input("–í–≤–µ–¥–∏—Ç–µ –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ —Å—Ç–æ–ª–±—Ü–æ–≤ N: "))

matrix = []
print(f"\n–í–≤–µ–¥–∏—Ç–µ –º–∞—Ç—Ä–∏—Ü—É {M}x{N} (–ø–æ {N} —á–∏—Å–µ–ª –≤ –∫–∞–∂–¥–æ–π —Å—Ç—Ä–æ–∫–µ):")

# –≤–≤–æ–¥ –º–∞—Ç—Ä–∏—Ü—ã 
for i in range(M):
    while True:
        try:
            row_input = input(f"–°—Ç—Ä–æ–∫–∞ {i+1}: ")
            row = list(map(int, row_input.split()))
            
            if len(row) != N:
                print(f"–û—à–∏–±–∫–∞! –ù—É–∂–Ω–æ –≤–≤–µ—Å—Ç–∏ —Ä–æ–≤–Ω–æ {N} —á–∏—Å–µ–ª. –ü–æ–ø—Ä–æ–±—É–π—Ç–µ —Å–Ω–æ–≤–∞.")
                continue
                
            matrix.append(row)
            break
        except ValueError:
            print("–û—à–∏–±–∫–∞! –í–≤–æ–¥–∏—Ç–µ —Ç–æ–ª—å–∫–æ —Ü–µ–ª—ã–µ —á–∏—Å–ª–∞. –ü–æ–ø—Ä–æ–±—É–π—Ç–µ —Å–Ω–æ–≤–∞.")

result, min_elements = find_max_of_mins(matrix)

# –≤—ã–≤–æ–¥ —Ä–µ–∑—É–ª—å—Ç–∞—Ç–æ–≤
print("\n")
print("–ú–∞—Ç—Ä–∏—Ü–∞:")
for i, row in enumerate(matrix):
    print(f"–°—Ç—Ä–æ–∫–∞ {i+1}: {row}")

print("\n–ê–Ω–∞–ª–∏–∑:")
for i, (row, min_val) in enumerate(zip(matrix, min_elements), 1):
    print(f"–°—Ç—Ä–æ–∫–∞ {i}: {row} ‚Üí –º–∏–Ω–∏–º–∞–ª—å–Ω—ã–π = {min_val}")


print(f"–ú–∏–Ω–∏–º–∞–ª—å–Ω—ã–µ —ç–ª–µ–º–µ–Ω—Ç—ã —Å—Ç—Ä–æ–∫: {min_elements}")
print(f"–ú–∞–∫—Å–∏–º–∞–ª—å–Ω—ã–π —Å—Ä–µ–¥–∏ –Ω–∏—Ö: {result}")
