'''Oпределение структуры с именем Student, содержащую
следующие поля: фамилия и инициалы; номер группы;
успеваемость – массив из пяти элементов.'''


# Структура Student
class Student:
    def __init__(self):
        self.name = ""          # фамилия и инициалы
        self.group = ""         # номер группы
        self.grades = []        # успеваемость (5 оценок)

# Ввод данных
student = Student()
print("Введите данные о студенте:\n")
student.name = input("Фамилия и инициалы:\n")
student.group = input("Номер группы: \n")

print("Введите 5 оценок через пробел (от 1 до 5):")
while True:
    grades_input = input().split()
    if len(grades_input) == 5:
        try:
            student.grades = [int(grade) for grade in grades_input]
            # Проверяем каждую оценку
            valid = True
            for grade in student.grades:
                if grade < 1 or grade > 5:
                    print("Оценка должна быть от 1 до 5")
                    valid = False
                    break
            if valid:
                break
        except ValueError:
            print("Введите только числа!")
    else:
        print("Нужно ввести ровно 5 оценок")

# Вывод данных
print("\nДанные студента:")
print(f"ФИО: {student.name}")
print(f"Группа: {student.group}")
print(f"Оценки: {student.grades}")
