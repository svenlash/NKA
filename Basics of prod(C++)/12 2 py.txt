'''Программа формирования информации о конкретном рейсе на поезде'''

def input_train():
    print("\nВведите данные о поезде:")
    
    # Создаем словарь сразу с вводом
    destination = input("Пункт назначения: ")
    
    while True:
        try:
            train_number = int(input("Номер поезда: "))
            if train_number > 0:
                break
            print("Номер должен быть больше 0")
        except:
            print("Введите число!")
    
    departure_time = input("Время отправления (ЧЧ:ММ): ")
    
    # Возвращаем готовый словарь
    return {
        'destination': destination,
        'train_number': train_number,
        'departure_time': departure_time
    }

def output_train(train):
    print(f"\nПоезд №{train['train_number']}")
    print(f"Направление: {train['destination']}")
    print(f"Отправление: {train['departure_time']}")

# Основная программа
print("Программа формирования структуры Train")
train = input_train()  # Теперь функция сама создает и возвращает словарь
output_train(train)
