'''Программа формирования информации о конкретном рейсе на поезде'''

def input_train()
    Функция для ввода данных о поезде
    print(nВведите данные о поезде)
    
    destination = input(Пункт назначения )
    
    while True
        try
            train_number = int(input(Номер поезда ))
            if train_number  0
                break
            print(Номер должен быть больше 0)
        except
            print(Введите число!)
    
    departure_time = input(Время отправления (ЧЧММ) )
    
    return {
        'destination' destination,
        'train_number' train_number,
        'departure_time' departure_time
    }

def output_train(train)
    Функция для вывода данных о поезде
    print(fПоезд №{train['train_number']4d}  
          fНаправление {train['destination']20}  
          fОтправление {train['departure_time']})

def sort_trains_by_time(trains_list)
    
    Сортировка списка поездов по времени отправления
    
    return sorted(trains_list, key=lambda x x['departure_time'])
